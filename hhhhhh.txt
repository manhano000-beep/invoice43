<script>
let total = 0;

// إضافة منتج
function addItem() {
    let item = document.getElementById("item").value;
    let price = parseFloat(document.getElementById("price").value);
    let qty = parseFloat(document.getElementById("qty").value);

    if (!item || !price || !qty) {
        alert("أدخل كل البيانات");
        return;
    }

    let sum = price * qty;
    total += sum;

    let row = `
        <tr>
            <td>${item}</td>
            <td>${price}</td>
            <td>${qty}</td>
            <td>${sum}</td>
        </tr>
    `;

    document.getElementById("invoiceBody").innerHTML += row;
    document.getElementById("total").innerText = total;

    document.getElementById("item").value = "";
    document.getElementById("price").value = "";
    document.getElementById("qty").value = "";
}

// طباعة
function printInvoice() {
    updateExtraData();
    window.print();
}

// تحديث البيانات الاختيارية
function updateExtraData() {
    let company = document.getElementById("companyName").value.trim();
    let agent = document.getElementById("agentName").value.trim();

    let html = "";

    if (company !== "") {
        html += "<p><strong>اسم المؤسسة:</strong> " + company + "</p>";
    }

    if (agent !== "") {
        html += "<p><strong>اسم المخلص:</strong> " + agent + "</p>";
    }

    document.getElementById("extraData").innerHTML = html;
}

// تحديث مباشر أثناء الكتابة
document.getElementById("companyName").addEventListener("input", updateExtraData);
document.getElementById("agentName").addEventListener("input", updateExtraData);

// قبل الطباعة
window.onbeforeprint = updateExtraData;
</script>
